# Fuzzer M2a Implementation Plan — State-Prerequisite Seeding

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Eliminate the `tecUNFUNDED_OFFER` failures seen in the M1 smoke (38/100 offers failed) by pre-seeding a dense trust-line + IOU-balance mesh between every pair of pool accounts before the fuzz loop starts.

**Architecture:** A new `SetupState` function in the `accounts` package runs between `FundFromGenesis` and the fuzz loop. Two phases: (1) for every ordered pair of distinct pool accounts, submit a TrustSet so holder trusts issuer for USD, wait for validation; (2) for every ordered pair, issuer sends a small USD Payment to the holder to fund the trust line with actual IOU balance. After this, any OfferCreate submitted during the fuzz loop that sells USD on one pool account's books from another's balance has backing liquidity.

**Tech Stack:** Go 1.24, existing `rpcclient` extended with an IOU-payment helper, existing `runners.Config` extended with a `SkipSetup` flag.

**Branch:** `fuzz/m2a` stacked on top of `fuzz/m1` (M1 is in review at PR #8).

**Scope:**
- In: `SetupState` submits the mesh, waits for validation between phases; `SubmitPaymentIOU` RPC helper; `runners.Config.SkipSetup`; runner wires it in; new smoke validates the fix.
- Out (deferred to M2b+): state tracking for tx types beyond OfferCreate; currencies beyond USD; sparse-mesh optimization; amendment-aware setup (e.g., only run if the current ledger has no trust lines already — not needed because each fuzz run starts from genesis).

// captureServer records each /submit body it receives. The `tx_json.TransactionType`
// of every request is stored for later counting. Payments with a map Amount are
// classified as "PaymentIOU" so we can distinguish them from TrustSet.
type capturedCall struct {
	TxType string
	IsIOU  bool
}

func newCaptureServer(results string) (*httptest.Server, *sync.Mutex, *[]capturedCall) {
	var mu sync.Mutex
	calls := []capturedCall{}
	srv := httptest.NewServer(http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
		body, _ := io.ReadAll(r.Body)
		var req map[string]any
		_ = json.Unmarshal(body, &req)

		call := capturedCall{}
		if params, ok := req["params"].([]any); ok && len(params) > 0 {
			if p, ok := params[0].(map[string]any); ok {
				if txj, ok := p["tx_json"].(map[string]any); ok {
					call.TxType, _ = txj["TransactionType"].(string)
					if _, ok := txj["Amount"].(map[string]any); ok {
						call.IsIOU = true
					}
				}
			}
		}
		mu.Lock()
		calls = append(calls, call)
		mu.Unlock()

		_, _ = w.Write([]byte(results))
	}))
	return srv, &mu, &calls
}

func TestSetupState_SubmitsMeshAndIOUFunding(t *testing.T) {
	srv, mu, calls := newCaptureServer(`{"result":{"engine_result":"tesSUCCESS","engine_result_code":0,"engine_result_message":"","tx_json":{"hash":"AAAA"},"status":"success"}}`)
	defer srv.Close()

	pool, err := NewPool(0xabc, 4)
	if err != nil {
		t.Fatal(err)
	}
	client := rpcclient.New(srv.URL)

	// Override the ledger-wait so the test doesn't take 10s.
	orig := SetupLedgerWait
	SetupLedgerWait = 10 * time.Millisecond
	defer func() { SetupLedgerWait = orig }()

	if err := SetupState(client, pool); err != nil {
		t.Fatalf("SetupState: %v", err)
	}

	mu.Lock()
	defer mu.Unlock()

	// For n=4, we expect n*(n-1) = 12 TrustSets + 12 IOU Payments = 24 calls.
	var trustSet, iouPay int
	for _, c := range *calls {
		switch {
		case c.TxType == "TrustSet":
			trustSet++
		case c.TxType == "Payment" && c.IsIOU:
			iouPay++
		}
	}
	if trustSet != 12 {
		t.Fatalf("TrustSet count = %d, want 12", trustSet)
	}
	if iouPay != 12 {
		t.Fatalf("IOU Payment count = %d, want 12", iouPay)
	}

	// TrustSets must all precede IOU payments (phase ordering).
	// Find the index of the last TrustSet and the first IOU Payment.
	lastTrust, firstIOU := -1, -1
	for i, c := range *calls {
		if c.TxType == "TrustSet" {
			lastTrust = i
		}
		if c.TxType == "Payment" && c.IsIOU && firstIOU < 0 {
			firstIOU = i
		}
	}
	if !(lastTrust < firstIOU) {
		t.Fatalf("phase ordering violated: lastTrust=%d firstIOU=%d", lastTrust, firstIOU)
	}
}

```go
package accounts

import (
	"fmt"
	"time"

	"github.com/XRPL-Commons/xrpl-confluence/sidecar/internal/rpcclient"
)

// Setup* constants govern the dense trust-line mesh seeded before fuzzing.
// They are package variables (not consts) so tests can shorten SetupLedgerWait.
var (
	SetupCurrency   = "USD"
	SetupLimit      = "1000000" // trust-line limit in IOU units
	SetupIOUFunding = "10000"   // amount of IOU each holder receives from each issuer
	SetupLedgerWait = 5 * time.Second
)

// SetupState seeds a dense trust-line + IOU-balance mesh between every pair
// of distinct pool accounts. After this returns, any OfferCreate selling USD
// that a pool account received from another pool account has liquidity —
// eliminating the tecUNFUNDED_OFFER failures observed in the M1 smoke.
//
// Phases (each followed by a wait so the prior phase's txs are validated):
//   1. For every ordered pair (holder=i, issuer=j), i != j:
//      holder TrustSets issuer for USD up to SetupLimit.
//   2. For every ordered pair (holder=i, issuer=j), i != j:
//      issuer sends a Payment of SetupIOUFunding USD to holder.
//
// Totals for pool size n: 2 * n * (n-1) submissions. The linear submission
// pattern matches the existing FundFromGenesis helper — parallelism is a
// future optimization if setup becomes a bottleneck.
func SetupState(client *rpcclient.Client, pool *Pool) error {
	wallets := pool.All()
	n := len(wallets)
	if n < 2 {
		return nil
	}

	// Phase 1: trust-line mesh.
	for i := 0; i < n; i++ {
		for j := 0; j < n; j++ {
			if i == j {
				continue
			}
			res, err := client.SubmitTrustSet(
				wallets[i].Seed,
				wallets[i].ClassicAddress,
				SetupCurrency,
				wallets[j].ClassicAddress,
				SetupLimit,
			)
			if err != nil {
				return fmt.Errorf("trustset %s→%s: %w",
					wallets[i].ClassicAddress, wallets[j].ClassicAddress, err)
			}
			if res.EngineResult != "tesSUCCESS" && res.EngineResult != "terQUEUED" {
				return fmt.Errorf("trustset %s→%s: engine=%s (%s)",
					wallets[i].ClassicAddress, wallets[j].ClassicAddress,
					res.EngineResult, res.EngineResultMessage)
			}
		}
	}
	time.Sleep(SetupLedgerWait)

	// Phase 2: IOU funding. Issuer wallet[j] sends USD to holder wallet[i].
	for i := 0; i < n; i++ {
		for j := 0; j < n; j++ {
			if i == j {
				continue
			}
			amount := map[string]any{
				"currency": SetupCurrency,
				"issuer":   wallets[j].ClassicAddress,
				"value":    SetupIOUFunding,
			}
			res, err := client.SubmitPaymentIOU(
				wallets[j].Seed,
				wallets[j].ClassicAddress,
				wallets[i].ClassicAddress,
				amount,
			)
			if err != nil {
				return fmt.Errorf("iou payment %s→%s: %w",
					wallets[j].ClassicAddress, wallets[i].ClassicAddress, err)
			}
			if res.EngineResult != "tesSUCCESS" && res.EngineResult != "terQUEUED" {
				return fmt.Errorf("iou payment %s→%s: engine=%s (%s)",
					wallets[j].ClassicAddress, wallets[i].ClassicAddress,
					res.EngineResult, res.EngineResultMessage)
			}
		}
	}
	time.Sleep(SetupLedgerWait)
	return nil
}
```

## Post-M2a Follow-ups (not in scope, noted here so they aren't lost)

- **Sparse-mesh optimization.** For pool sizes > 20, the dense mesh becomes expensive (O(n²) txs). A sparse variant (e.g., pick 5 designated issuers, every account trusts those) gives similar coverage at O(n) cost.
- **Multi-currency.** Right now only USD. Easy extension once M2b lands — iterate over a currency list.
- **Background validation waiting.** Replace `time.Sleep(SetupLedgerWait)` with a poll on `ServerInfo().Validated.Seq` advancing. Cleaner than a fixed sleep.
- **Pool-of-1 warning.** Currently a pool of 1 silently skips setup. When M2b requires pool ≥ 2 anyway, remove this branch.
